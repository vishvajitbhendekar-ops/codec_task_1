import React, { useState } from "react"; import { Card, CardContent } from "@/components/ui/card"; import { Button } from "@/components/ui/button"; import { Input } from "@/components/ui/input"; import { Select, SelectItem } from "@/components/ui/select"; import { PieChart, Pie, Tooltip, BarChart, Bar, XAxis, YAxis } from "recharts";

export default function FinanceApp() { const [expenses, setExpenses] = useState([]); const [amount, setAmount] = useState(""); const [category, setCategory] = useState("Food");

const addExpense = () => { if (!amount) return; setExpenses([...expenses, { category, amount: Number(amount) }]); setAmount(""); };

const categoryData = Object.values( expenses.reduce((acc, e) => { acc[e.category] = acc[e.category] || { name: e.category, value: 0 }; acc[e.category].value += e.amount; return acc; }, {}) );

return ( <div className="p-6 grid gap-6 max-w-4xl mx-auto"> <h1 className="text-2xl font-bold">Personal Finance Manager</h1>

<Card>
    <CardContent className="p-4 grid gap-3">
      <h2 className="font-semibold">Add Expense</h2>
      <Input
        placeholder="Amount"
        value={amount}
        onChange={(e) => setAmount(e.target.value)}
      />
      <Select value={category} onValueChange={setCategory}>
        <SelectItem value="Food">Food</SelectItem>
        <SelectItem value="Travel">Travel</SelectItem>
        <SelectItem value="Shopping">Shopping</SelectItem>
        <SelectItem value="Bills">Bills</SelectItem>
      </Select>
      <Button onClick={addExpense}>Add</Button>
    </CardContent>
  </Card>

  <Card>
    <CardContent className="p-4">
      <h2 className="font-semibold mb-2">Expense Distribution</h2>
      <PieChart width={300} height={300}>
        <Pie data={categoryData} dataKey="value" nameKey="name" />
        <Tooltip />
      </PieChart>
    </CardContent>
  </Card>

  <Card>
    <CardContent className="p-4">
      <h2 className="font-semibold mb-2">Expense Analytics</h2>
      <BarChart width={400} height={250} data={categoryData}>
        <XAxis dataKey="name" />
        <YAxis />
        <Tooltip />
        <Bar dataKey="value" />
      </BarChart>
    </CardContent>
  </Card>
</div>

); }
